41const JOB_PERSIST_REQUESTED_EVENT: &str = "jobs.persist.requested";
42
43fn main() {
44    ensure_posix_or_exit();
45    print_startup_banner();
46    let app_started_at = Utc::now();
47
48    let runtime_args = std::env::args().skip(1).collect::<Vec<_>>();
49    let (runtime_flags, config_args) = self_debug::extract_runtime_flags(runtime_args);
50    let mut app_config = config::load_or_exit(config_args);
51    if runtime_flags.enabled {
52        app_config.storage.path = self_debug::resolve_storage_path(&app_config.storage);
53    }
54    let log_level = if runtime_flags.enabled {
55        LogLevel::Verbose
56    } else {
57        LogLevel::from_config_value(&app_config.logging.level).unwrap_or_else(|| {
58            eprintln!(
59                "invalid logging.level '{}'. Allowed values: error, warn, info, debug, verbose",
60                app_config.logging.level
61            );
62            process::exit(2);
63        })
64    };
65
66    let logger = Arc::new(Logger::new(LoggerConfig {
67        min_level: log_level,
68        human_friendly: app_config.logging.human_friendly,
69    }));
70    let storage = Arc::new(measure_execution(
71        "storage.initialize",
72        Some("main::startup"),
73        logger.as_ref(),
74        || StorageFacade::initialize(&app_config, logger.as_ref()),
75    )
76    .unwrap_or_else(|error| {
77        eprintln!("storage initialization error: {error}");
78        process::exit(2);
79    }));
80    let self_debug_artifacts_path: Option<PathBuf> = if runtime_flags.enabled {
81        Some(storage.data_path().to_path_buf())
82    } else {
83        None
84    };
85    let mut persistent_queues =
86        PersistentQueuePool::bootstrap(storage.as_ref(), logger.as_ref()).unwrap_or_else(|error| {
87            eprintln!("queue bootstrap error: {error}");
88            process::exit(2);
89        });
90    let jobs_pool = Arc::new(Mutex::new(JobsPool::new()));
91    logger.log(
92        LogLevel::Info,
93        Some("main::orchestrator"),
94        "Queue pool bootstrapped from persistence",
95        Some(json!({
96            "queues_count": persistent_queues.queue_pool().snapshot().queues.len(),
97            "includes_system_queue": persistent_queues.queue_pool().get_queue("_system").is_some()
98        })),
99    );
100    let server = TcpServer::from_app_config(&app_config).unwrap_or_else(|error| {
101        eprintln!("server startup error: {error}");
102        process::exit(2);
103    });
104    let bound_addr = server.local_addr().unwrap_or_else(|error| {
105        eprintln!("server startup error: failed to read local address: {error}");
106        process::exit(2);
107    });
108    logger.log(
109        LogLevel::Info,
110        Some("main::server"),
111        &format!(
112            "{} v{} started non-blocking TCP server",
113            env!("CARGO_PKG_NAME"),
114            env!("CARGO_PKG_VERSION")
115        ),
116        Some(json!({
117            "bind_address": bound_addr.to_string(),
118            "host": app_config.server.host,
119            "port": app_config.server.port,
120            "tls_enabled": app_config.server.tls_enabled
121        })),
122    );
123    if !app_config.server.tls_enabled {
124        logger.info(
125            Some("main::server"),
126            "TLS is disabled for now; TCP connections are plain and persistent full-duplex",
127        );
128    }
129    let wire_codec = Arc::new(WireCodec::from_app_config(&app_config).unwrap_or_else(|error| {
130        eprintln!("wire codec configuration error: {error}");
131        process::exit(2);
132    }));
133    logger.log(
134        LogLevel::Info,
135        Some("main::wire"),
136        "Wire codec initialized",
137        Some(json!({
138            "max_envelope_size_bytes": wire_codec.max_envelope_size_bytes()
139        })),
140    );
141
142    let emitter = Arc::new(EventEmitter::new());
143    let pools = Arc::new(ConnectionWorkerPools::new());
144    let timeout_policy = AnonymousTimeoutPolicy {
145        unhelloed_max_lifetime_seconds: app_config
146            .wire
147            .session
148            .unhelloed_max_lifetime_seconds,
149        helloed_unregistered_max_lifetime_seconds: app_config
150            .wire
151            .session
152            .helloed_unregistered_max_lifetime_seconds,
153        ident_register_timeout_seconds: app_config.wire.session.ident_register_timeout_seconds,
154    };
155
156    emitter.on("app.started", |event| {
157        println!("sync event observed: {}", event.name);
158        Ok(())
159    });
160    emitter.on_async("app.started", |event| {
161        eprintln!("async event observed: {}", event.name);
162        Ok(())
163    });
164    let storage_for_job_persist = Arc::clone(&storage);
165    let jobs_pool_for_persist = Arc::clone(&jobs_pool);
166    let logger_for_job_persist = Arc::clone(&logger);
167    emitter.on(JOB_PERSIST_REQUESTED_EVENT, move |event| {
168        let Some(payload) = event.payload.as_ref() else {
169            return Err("job persist event payload is missing".to_owned());
170        };
171
172        let Some(job_uuid_raw) = payload.get("job_uuid").and_then(serde_json::Value::as_str) else {
173            return Err("job persist event missing string 'job_uuid'".to_owned());
174        };
175        let job_uuid = uuid::Uuid::parse_str(job_uuid_raw)
176            .map_err(|_| "job persist event has invalid 'job_uuid'".to_owned())?;
177        let Some(record) = payload.get("record") else {
178            return Err("job persist event missing 'record'".to_owned());
179        };
180        let Some(execution_start_ms) = payload
181            .get("execution_start_ms")
182            .and_then(serde_json::Value::as_i64)
183        else {
184            return Err("job persist event missing i64 'execution_start_ms'".to_owned());
185        };
186        let Some(created_at_ms) = payload
187            .get("created_at_ms")
188            .and_then(serde_json::Value::as_i64)
189        else {
190            return Err("job persist event missing i64 'created_at_ms'".to_owned());
191        };
192        let Some(queue_name) = payload.get("queue_name").and_then(serde_json::Value::as_str) else {
193            return Err("job persist event missing string 'queue_name'".to_owned());
194        };
195        let Some(status) = payload.get("status").and_then(serde_json::Value::as_str) else {
196            return Err("job persist event missing string 'status'".to_owned());
197        };
198
199        storage_for_job_persist
200            .upsert_job_record(job_uuid, record, execution_start_ms, created_at_ms, queue_name, status)
201            .map_err(|error| error.to_string())?;
202
203        let mut guard = jobs_pool_for_persist
204            .lock()
205            .map_err(|_| "jobs pool lock poisoned while removing persisted job".to_owned())?;
206        let _ = guard.remove_job(job_uuid);
207
208        logger_for_job_persist.debug(
209            Some("main::jobs"),
210            &format!("Persisted staged job '{job_uuid}' and removed it from transient pool"),
211        );
212        Ok(())
213    });
214    let pools_for_maintenance = Arc::clone(&pools);
215    let wire_codec_for_maintenance = Arc::clone(&wire_codec);
216    let logger_for_maintenance = Arc::clone(&logger);
217    let storage_for_status_progression = Arc::clone(&storage);
218    let logger_for_status_progression = Arc::clone(&logger);
219    let maintenance_interval_seconds = timeout_policy.maintenance_interval_seconds();
220    let heartbeat_maintenance_last_run = Arc::new(Mutex::new(None::<DateTime<Utc>>));
221    let heartbeat_maintenance_last_run_for_listener = Arc::clone(&heartbeat_maintenance_last_run);
222    emitter.on(HEARTBEAT_EVENT, move |_| {
223        let now = Utc::now();
224        let mut guard = heartbeat_maintenance_last_run_for_listener
225            .lock()
226            .map_err(|_| "heartbeat maintenance lock poisoned".to_owned())?;
227        if let Some(last_run) = *guard {
228            if (now - last_run).num_seconds() < maintenance_interval_seconds as i64 {
229                return Ok(());
230            }
231        }
232        *guard = Some(now);
233        purge_stale_anonymous_connections(
234            pools_for_maintenance.as_ref(),
235            wire_codec_for_maintenance.as_ref(),
236            logger_for_maintenance.as_ref(),
237            timeout_policy,
238        );
239        Ok(())
240    });
241    emitter.on(HEARTBEAT_EVENT, move |_| {
242        advance_persisted_jobs_statuses(
243            storage_for_status_progression.as_ref(),
244            logger_for_status_progression.as_ref(),
245        )
246        .map_err(|error| format!("jobs status heartbeat progression failed: {error}"))
247    });
248
249    emitter.emit_or_exit("app.started", Some(json!({"component":"main"})));
250
251    let mut heartbeat = Heartbeat::from_app_config(Arc::clone(&emitter), &app_config)
252        .expect("heartbeat configuration should be valid");
253    logger.log(
254        LogLevel::Info,
255        Some("main::heartbeat"),
256        "Heartbeat initialized",
257        Some(heartbeat.initial_metadata_payload()),
258    );
259    heartbeat.start().expect("heartbeat should start");
260
261    let shutdown_hooks = ShutdownHooks::install().unwrap_or_else(|error| {
262        eprintln!("failed to install shutdown hooks: {error}");
263        process::exit(2);
264    });
265    logger.info(
266        Some("main::shutdown"),
267        "Shutdown hooks installed for SIGINT/SIGTERM",
268    );
269
270    let self_debug_runner = if runtime_flags.enabled {
271        Some(self_debug::spawn_runner(bound_addr, (*wire_codec).clone()))
272    } else {
273        None
274    };
275    let mut self_debug_result: Option<Result<(), self_debug::SelfDebugError>> = None;
276
277    while !shutdown_hooks.is_triggered() {
278        if let Some(connection) = server.try_accept_persistent().unwrap_or_else(|error| {
279            eprintln!("server accept error: {error}");
280            process::exit(2);
281        }) {
282            let connection_id = pools.register_anonymous(Arc::clone(&connection));
283            let anon_metadata = pools
284                .anonymous
285                .snapshot(connection_id)
286                .expect("anonymous connection should be present right after registration");
287            logger.log(
288                LogLevel::Info,
289                Some("main::server"),
290                "Accepted persistent full-duplex TCP connection",
291                Some(json!({
292                    "connection_id": connection.id(),
293                    "anonymous_connection_id": connection_id,
294                    "peer_addr": connection.peer_addr().to_string(),
295                    "connected_at": anon_metadata.connected_at.to_rfc3339(),
296                    "helloed_at": anon_metadata.helloed_at.map(|v| v.to_rfc3339()),
297                    "ident_reply_deadline_at": anon_metadata.ident_reply_deadline_at.map(|v| v.to_rfc3339()),
298                    "tls_enabled": app_config.server.tls_enabled
299                })),
300            );
301        }
302        process_anonymous_client_messages(
303            pools.as_ref(),
304            wire_codec.as_ref(),
305            logger.as_ref(),
306        );
307        process_worker_client_messages(
308            pools.as_ref(),
309            emitter.as_ref(),
310            wire_codec.as_ref(),
311            logger.as_ref(),
312            storage.as_ref(),
313            &mut persistent_queues,
314            jobs_pool.as_ref(),
315            app_config.pagination.page_size,
316            app_started_at,
317        );
318
319        if let Some(rx) = &self_debug_runner {
320            match rx.try_recv() {
321                Ok(result) => {
322                    self_debug_result = Some(result);
323                    break;
324                }
325                Err(TryRecvError::Empty) => {}
326                Err(TryRecvError::Disconnected) => {
327                    self_debug_result = Some(Err(self_debug::SelfDebugError::Io(
328                        io::Error::other("self-debug runner disconnected"),
329                    )));
330                    break;
331                }
332            }
333        }
334        thread::sleep(Duration::from_millis(100));
335    }
336
337    logger.info(
338        Some("main::shutdown"),
339        "Shutdown signal received, starting graceful shutdown",
340    );
341    emitter.begin_shutdown();
342    heartbeat.stop().expect("heartbeat should stop");
343    if let Err(error) = persistent_queues.persist_current_state(storage.as_ref()) {
344        eprintln!("queue persistence error during shutdown: {error}");
345        process::exit(2);
346    }
347    server.shutdown_all_connections();
348    drain_transient_jobs_pool_before_exit(storage.as_ref(), jobs_pool.as_ref(), logger.as_ref());
349    thread::sleep(Duration::from_millis(100));
350
351    let drained = emitter.wait_for_idle(Duration::from_secs(3));
352    if drained {
353        logger.info(
354            Some("main::shutdown"),
355            "All running listeners completed before timeout",
356        );
357    } else {
358        logger.warn(
359            Some("main::shutdown"),
360            "Listener drain timeout reached; continuing shutdown",
361        );
362    }
363
364    drop(server);
365    drop(persistent_queues);
366    drop(storage);
367    let _wire_codec = wire_codec;
368
369    let mut self_debug_cleanup_error: Option<self_debug::SelfDebugError> = None;
370    if !runtime_flags.keep_artifacts {
371        if let Some(path) = self_debug_artifacts_path {
372            if let Err(error) = self_debug::cleanup_artifacts(path.as_path()) {
373                self_debug_cleanup_error = Some(error);
374            }
375        }
376    }
377
378    if let Some(Err(error)) = self_debug_result {
379        if let Some(cleanup_error) = self_debug_cleanup_error {
380            eprintln!("self-debug cleanup error (after run failure): {cleanup_error}");
381        }
382        eprintln!("self-debug run failed: {error}");
383        process::exit(2);
384    }
385    if let Some(cleanup_error) = self_debug_cleanup_error {
386        eprintln!("self-debug cleanup error: {cleanup_error}");
387        process::exit(2);
388    }
389
390    logger.info(
391        Some("main::shutdown"),
392        "TCP server stopped and shutdown completed",
393    );
394}
