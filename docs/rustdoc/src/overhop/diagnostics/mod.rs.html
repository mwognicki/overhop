13#[derive(Debug)]
14pub enum DiagnosticsError {
15    Storage(StorageError),
16    Io(std::io::Error),
17}
18
19impl fmt::Display for DiagnosticsError {
20    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
21        match self {
22            Self::Storage(source) => write!(f, "storage diagnostics error: {source}"),
23            Self::Io(source) => write!(f, "io diagnostics error: {source}"),
24        }
25    }
26}
27
28impl std::error::Error for DiagnosticsError {}
29
30impl From<StorageError> for DiagnosticsError {
31    fn from(value: StorageError) -> Self {
32        Self::Storage(value)
33    }
34}
35
36impl From<std::io::Error> for DiagnosticsError {
37    fn from(value: std::io::Error) -> Self {
38        Self::Io(value)
39    }
40}

42pub fn build_status_payload(
43    app_started_at: DateTime<Utc>,
44    pools: &ConnectionWorkerPools,
45    storage: &StorageFacade,
46    queue_pool: &QueuePool,
47) -> Result<PayloadMap, DiagnosticsError> {
48    let mut payload = PayloadMap::new();
49    payload.insert(
50        "application".to_owned(),
51        application_status(app_started_at, Utc::now()),
52    );
53    payload.insert("memory".to_owned(), memory_stats());
54    payload.insert("pools".to_owned(), pool_stats(pools));
55    payload.insert("storage".to_owned(), storage_summary(storage)?);
56    payload.insert("queues".to_owned(), queue_pool_stats(queue_pool));
57    Ok(payload)
58}

94pub fn memory_stats() -> Value {
95    let mut vm_rss_kb: Option<i64> = None;
96    let mut vm_size_kb: Option<i64> = None;
97    let mut vm_peak_kb: Option<i64> = None;
98
99    if let Ok(status) = fs::read_to_string("/proc/self/status") {
100        for line in status.lines() {
101            if line.starts_with("VmRSS:") {
102                vm_rss_kb = parse_kb_field(line);
103            } else if line.starts_with("VmSize:") {
104                vm_size_kb = parse_kb_field(line);
105            } else if line.starts_with("VmPeak:") {
106                vm_peak_kb = parse_kb_field(line);
107            }
108        }
109    }
110
111    Value::Map(vec![
112        (
113            Value::String("source".into()),
114            Value::String("/proc/self/status".into()),
115        ),
116        (
117            Value::String("available".into()),
118            Value::Boolean(vm_rss_kb.is_some() || vm_size_kb.is_some() || vm_peak_kb.is_some()),
119        ),
120        (
121            Value::String("vm_rss_kb".into()),
122            vm_rss_kb.map_or(Value::Nil, |v| Value::Integer(v.into())),
123        ),
124        (
125            Value::String("vm_size_kb".into()),
126            vm_size_kb.map_or(Value::Nil, |v| Value::Integer(v.into())),
127        ),
128        (
129            Value::String("vm_peak_kb".into()),
130            vm_peak_kb.map_or(Value::Nil, |v| Value::Integer(v.into())),
131        ),
132    ])
133}

135pub fn pool_stats(pools: &ConnectionWorkerPools) -> Value {
136    let anonymous = pools.anonymous.maintenance_snapshot();
137    let workers = pools.workers.active_connections();
138    let mut subscriptions_total = 0_i64;
139
140    for (worker_id, _) in &workers {
141        if let Some(snapshot) = pools.workers.snapshot(*worker_id) {
142            subscriptions_total += snapshot.metadata.subscriptions.len() as i64;
143        }
144    }
145
146    let helloed_anonymous = anonymous
147        .iter()
148        .filter(|entry| entry.helloed_at.is_some())
149        .count() as i64;
150    let ident_challenged = anonymous
151        .iter()
152        .filter(|entry| entry.ident_reply_deadline_at.is_some())
153        .count() as i64;
154
155    Value::Map(vec![
156        (
157            Value::String("anonymous_total".into()),
158            Value::Integer((anonymous.len() as i64).into()),
159        ),
160        (
161            Value::String("anonymous_helloed".into()),
162            Value::Integer(helloed_anonymous.into()),
163        ),
164        (
165            Value::String("anonymous_ident_challenged".into()),
166            Value::Integer(ident_challenged.into()),
167        ),
168        (
169            Value::String("workers_total".into()),
170            Value::Integer((workers.len() as i64).into()),
171        ),
172        (
173            Value::String("worker_subscriptions_total".into()),
174            Value::Integer(subscriptions_total.into()),
175        ),
176    ])
177}

179pub fn storage_summary(storage: &StorageFacade) -> Result<Value, DiagnosticsError> {
180    let persisted_queues = storage.load_queues()?;
181    let data_path = storage.data_path();
182    let path_exists = data_path.exists();
183    let total_size_bytes = if path_exists {
184        directory_size_bytes(data_path)?
185    } else {
186        0
187    };
188
189    Ok(Value::Map(vec![
190        (
191            Value::String("engine".into()),
192            Value::String(storage.engine().as_str().into()),
193        ),
194        (
195            Value::String("data_path".into()),
196            Value::String(data_path.display().to_string().into()),
197        ),
198        (
199            Value::String("path_exists".into()),
200            Value::Boolean(path_exists),
201        ),
202        (
203            Value::String("persisted_queues_count".into()),
204            Value::Integer((persisted_queues.len() as i64).into()),
205        ),
206        (
207            Value::String("data_size_bytes".into()),
208            Value::Integer((total_size_bytes as i64).into()),
209        ),
210    ]))
211}

213pub fn queue_pool_stats(queue_pool: &QueuePool) -> Value {
214    let snapshot = queue_pool.snapshot();
215    let total = snapshot.queues.len() as i64;
216    let paused = snapshot
217        .queues
218        .iter()
219        .filter(|queue| queue.is_paused())
220        .count() as i64;
221    let active = total - paused;
222
223    Value::Map(vec![
224        (
225            Value::String("bootstrapped".into()),
226            Value::Boolean(snapshot.bootstrapped),
227        ),
228        (
229            Value::String("total".into()),
230            Value::Integer(total.into()),
231        ),
232        (
233            Value::String("active".into()),
234            Value::Integer(active.into()),
235        ),
236        (
237            Value::String("paused".into()),
238            Value::Integer(paused.into()),
239        ),
240    ])
241}

243fn parse_kb_field(line: &str) -> Option<i64> {
244    line.split_whitespace().nth(1)?.parse::<i64>().ok()
245}

247fn directory_size_bytes(path: &Path) -> Result<u64, std::io::Error> {
248    let mut total = 0_u64;
249    for entry in fs::read_dir(path)? {
250        let entry = entry?;
251        let metadata = entry.metadata()?;
252        if metadata.is_file() {
253            total = total.saturating_add(metadata.len());
254        } else if metadata.is_dir() {
255            total = total.saturating_add(directory_size_bytes(&entry.path())?);
256        }
257    }
258    Ok(total)
259}
