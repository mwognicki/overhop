41#[derive(Debug)]
42pub enum ServerError {
43    Bind {
44        address: String,
45        source: io::Error,
46    },
47    SetNonBlocking {
48        source: io::Error,
49    },
50    TlsNotSupportedYet,
51    ConfigureAcceptedStream {
52        source: io::Error,
53    },
54    StreamClone {
55        source: io::Error,
56    },
57}
58
59impl fmt::Display for ServerError {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        match self {
62            Self::Bind { address, source } => {
63                write!(f, "failed to bind TCP server on {address}: {source}")
64            }
65            Self::SetNonBlocking { source } => {
66                write!(f, "failed to set TCP server to non-blocking mode: {source}")
67            }
68            Self::TlsNotSupportedYet => {
69                write!(f, "TLS is enabled in config but TLS transport is not implemented yet")
70            }
71            Self::ConfigureAcceptedStream { source } => {
72                write!(f, "failed to configure accepted TCP stream: {source}")
73            }
74            Self::StreamClone { source } => {
75                write!(f, "failed to clone accepted TCP stream for full duplex IO: {source}")
76            }
77        }
78    }
79}

81impl std::error::Error for ServerError {}

83pub struct PersistentConnection {
84    id: u64,
85    peer_addr: SocketAddr,
86    reader: Mutex<TcpStream>,
87    writer: Mutex<TcpStream>,
88}
89
90impl PersistentConnection {
91    fn new(id: u64, stream: TcpStream, peer_addr: SocketAddr) -> Result<Self, ServerError> {
92        stream
93            .set_nodelay(true)
94            .map_err(|source| ServerError::ConfigureAcceptedStream { source })?;
95        stream
96            .set_nonblocking(true)
97            .map_err(|source| ServerError::ConfigureAcceptedStream { source })?;
98
99        let writer = stream
100            .try_clone()
101            .map_err(|source| ServerError::StreamClone { source })?;
102
103        Ok(Self {
104            id,
105            peer_addr,
106            reader: Mutex::new(stream),
107            writer: Mutex::new(writer),
108        })
109    }
110
111    pub fn id(&self) -> u64 {
112        self.id
113    }
114
115    pub fn peer_addr(&self) -> SocketAddr {
116        self.peer_addr
117    }
118
119    pub fn try_read(&self, buffer: &mut [u8]) -> io::Result<usize> {
120        self.reader
121            .lock()
122            .expect("connection reader lock poisoned")
123            .read(buffer)
124    }
125
126    pub fn try_write(&self, payload: &[u8]) -> io::Result<usize> {
127        self.writer
128            .lock()
129            .expect("connection writer lock poisoned")
130            .write(payload)
131    }
132
133    pub fn shutdown(&self) -> io::Result<()> {
134        let _ = self
135            .reader
136            .lock()
137            .expect("connection reader lock poisoned")
138            .shutdown(Shutdown::Both);
139        self.writer
140            .lock()
141            .expect("connection writer lock poisoned")
142            .shutdown(Shutdown::Both)
143    }
144}

146pub struct TcpServer {
147    listener: TcpListener,
148    next_connection_id: AtomicU64,
149    active_connections: Mutex<HashMap<u64, Arc<PersistentConnection>>>,
150}
151
152impl TcpServer {
153    pub fn bind(config: &ServerConfig) -> Result<Self, ServerError> {
154        if config.tls_enabled {
155            return Err(ServerError::TlsNotSupportedYet);
156        }
157
158        let address = format!("{}:{}", config.host, config.port);
159        let listener = TcpListener::bind(&address).map_err(|source| ServerError::Bind {
160            address,
161            source,
162        })?;
163        listener
164            .set_nonblocking(true)
165            .map_err(|source| ServerError::SetNonBlocking { source })?;
166
167        Ok(Self {
168            listener,
169            next_connection_id: AtomicU64::new(1),
170            active_connections: Mutex::new(HashMap::new()),
171        })
172    }
173
174    pub fn from_app_config(app_config: &config::AppConfig) -> Result<Self, ServerError> {
175        let cfg = ServerConfig::from(app_config.server.clone());
176        Self::bind(&cfg)
177    }
178
179    pub fn local_addr(&self) -> io::Result<SocketAddr> {
180        self.listener.local_addr()
181    }
182
183    pub fn try_accept(&self) -> io::Result<(TcpStream, SocketAddr)> {
184        self.listener.accept()
185    }
186
187    pub fn try_accept_persistent(&self) -> Result<Option<Arc<PersistentConnection>>, ServerError> {
188        match self.listener.accept() {
189            Ok((stream, peer_addr)) => {
190                let id = self.next_connection_id.fetch_add(1, Ordering::Relaxed);
191                let connection = Arc::new(PersistentConnection::new(id, stream, peer_addr)?);
192                self.active_connections
193                    .lock()
194                    .expect("active connections lock poisoned")
195                    .insert(id, Arc::clone(&connection));
196                Ok(Some(connection))
197            }
198            Err(source) if source.kind() == io::ErrorKind::WouldBlock => Ok(None),
199            Err(source) => Err(ServerError::ConfigureAcceptedStream { source }),
200        }
201    }
202
203    pub fn connection_count(&self) -> usize {
204        self.active_connections
205            .lock()
206            .expect("active connections lock poisoned")
207            .len()
208    }
209
210    pub fn drop_connection(&self, id: u64) {
211        self.active_connections
212            .lock()
213            .expect("active connections lock poisoned")
214            .remove(&id);
215    }
216
217    pub fn shutdown_all_connections(&self) {
218        let mut connections = self
219            .active_connections
220            .lock()
221            .expect("active connections lock poisoned");
222
223        for connection in connections.values() {
224            let _ = connection.shutdown();
225        }
226        connections.clear();
227    }
228}

283    #[test]
284    fn accepts_persistent_full_duplex_connection() {
285        let cfg = ServerConfig {
286            host: "127.0.0.1".to_owned(),
287            port: 0,
288            tls_enabled: false,
289        };
290        let server = TcpServer::bind(&cfg).expect("server should bind");
291        let addr = server.local_addr().expect("local addr should exist");
292
293        let client = TcpStream::connect(addr).expect("client should connect");
294        client
295            .set_nonblocking(true)
296            .expect("client should be nonblocking");
297
298        let mut accepted = None;
299        for _ in 0..20 {
300            if let Some(conn) = server
301                .try_accept_persistent()
302                .expect("accept poll should not fail")
303            {
304                accepted = Some(conn);
305                break;
306            }
307            thread::sleep(Duration::from_millis(10));
308        }
309
310        let conn = accepted.expect("server should accept connection");
311        assert_eq!(server.connection_count(), 1);
312
313        let write_result = conn.try_write(b"ping");
314        assert!(write_result.is_ok() || write_result.err().is_some());
315
316        let mut buf = [0_u8; 16];
317        let read_result = conn.try_read(&mut buf);
318        assert!(read_result.is_ok() || read_result.err().is_some());
319
320        server.drop_connection(conn.id());
321        assert_eq!(server.connection_count(), 0);
322    }
